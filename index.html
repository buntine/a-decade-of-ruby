            def sum(numbers)
              numbers.reduce(0) do |total, n|
                total + n
              end
            end
          

            def sum(numbers)
              numbers.reduce(0, :+)
            end
          

            numbers.reduce(:+)
          

            def sum_multiples(numbers, multiple)
              numbers
                .reject { |n| n % multiple > 0}
                .reduce(:+)
            end
