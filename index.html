            def sum_multiples(numbers, multiple)
              numbers
                .reject { |n| n % multiple > 0}
                .reduce(:+)
            end
          

            def sum_multiples(numbers, multiple)
              total = 0

              for n in numbers do
                if n % multiple > 0
                  total += n
                end
              end

              total
            end

            def excited(sentence, amount=1)
              "#{sentence}#{"!" * amount}"
            end
          

            sentence = "Awesome"
            puts excited(sentence, 2)  # "Awesome!!"
          

            class String
              def excited(amount=1)
                "#{self}#{"!" * amount}"
              end
            end
          

            puts "Awesome".excited(2)  # "Awesome!!"
